# QUERY 1 : Simple triple pattern matching
# "List all mayors of New York."

PREFIX ex: <http://example.org/nyc/>

SELECT ?mayor
WHERE {
    ?mayor a ex:Mayor .
}

# QUERY 2 : Filtering using logical/numerical operators
# "Which mayors served after the year 2010?"

PREFIX ex: <http://example.org/nyc/>

SELECT ?mayor ?start
WHERE {
    ?mayor ex:startDate ?start .
    FILTER ( ?start > "2010-01-01"^^xsd:date )
}


# QUERY 3 : OPTIONAL usage
# "List all mayors with their start and end dates (even if end date is missing)."

PREFIX ex: <http://example.org/nyc/>

SELECT ?mayor ?start ?end
WHERE {
    ?mayor a ex:Mayor .
    OPTIONAL { ?mayor ex:startDate ?start . }
    OPTIONAL { ?mayor ex:endDate ?end . }
}


# QUERY 4 : Aggregation (COUNT)
# "How many mayors belonged to each political party?"

PREFIX ex: <http://example.org/nyc/>

SELECT ?party (COUNT(?mayor) AS ?totalMayors)
WHERE {
    ?mayor ex:partyAffiliation ?party .
}
GROUP BY ?party


# QUERY 5 : Graph traversal across *3 hops*
# "Find the successor of the successor of the successor of Rudy Giuliani"
# (3-hop traversal through ex:succeededBy)


PREFIX ex: <http://example.org/nyc/>

SELECT ?hop3
WHERE {
    ex:Rudy_Giuliani ex:succeededBy ?hop1 .
    ?hop1 ex:succeededBy ?hop2 .
    ?hop2 ex:succeededBy ?hop3 .
}
