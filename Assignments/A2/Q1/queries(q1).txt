PREFIX : <http://www.semanticweb.org/hafsa/ontologies/2025/10/wound-ontology/>

# Query 1 — All wounds with patient and healing stage
SELECT ?wound ?woundLabel ?patient ?patientName ?stage ?stageLabel
WHERE {
  ?wound rdf:type :Wound .
  ?wound rdfs:label ?woundLabel .
  ?wound :affects ?patient .
  ?patient :patientName ?patientName .
  ?wound :healingStage ?stage .
  ?stage rdfs:label ?stageLabel .
}

# Query 2 — Treatments responsible for each wound

PREFIX : <http://www.semanticweb.org/hafsa/ontologies/2025/10/wound-ontology/>

SELECT ?wound ?woundLabel ?treatment ?treatmentTitle
WHERE {
  ?treatment rdf:type :Treatment .
  ?treatment :needsCheck ?wound .
  ?wound rdfs:label ?woundLabel .
  ?treatment rdfs:label ?treatmentTitle .
}


# Query 3 — Wounds with severity, cause, and patient

PREFIX : <http://www.semanticweb.org/hafsa/ontologies/2025/10/wound-ontology/>

SELECT ?wound ?woundLabel ?severity ?cause ?causeInfo ?patient ?patientName
WHERE {
  ?wound rdf:type :Wound .
  ?wound rdfs:label ?woundLabel .
  ?wound :severityScore ?severity .
  ?wound :hasCause ?cause .
  ?cause :causeInfo ?causeInfo .
  ?wound :affects ?patient .
  ?patient :patientName ?patientName .
}
