//wayoflifeOPENMP
#include <stdio.h>  // For printf(), fscanf(), fopen()
#include <stdlib.h> // For malloc(), free()
#include <omp.h>    // OpenMP library

// Tile size for tiled processing
// Grid will be processed in blocks of 32x32
#define TILE 32

// Number of simulation generations
#define GENERATIONS 10

// ------------------------------------------------------------
// Function to count alive neighbors around a cell
// ------------------------------------------------------------
int count_neighbors(int **grid,
                    int rows,
                    int cols,
                    int x,
                    int y)
{
    // Stores number of alive neighbors
    int count = 0;

    // Loop from -1 to +1 around current cell
    // i controls row movement
    // j controls column movement

    for (int i = -1; i <= 1; i++)
    {
        for (int j = -1; j <= 1; j++)
        {
            // Skip the current cell itself
            // because it is not its own neighbor
            if (i == 0 && j == 0)
                continue;

            // Calculate neighbor coordinates
            int nx = x + i;
            int ny = y + j;

            // Boundary checking
            // Prevents accessing invalid indices
            if (nx >= 0 && nx < rows &&
                ny >= 0 && ny < cols)
            {
                // Add neighbor value
                // Alive = 1
                // Dead = 0
                count += grid[nx][ny];
            }
        }
    }

    // Return total alive neighbors
    return count;
}

int main()
{
    // ------------------------------------------------------------
    // Open input file in read mode
    // ------------------------------------------------------------
    FILE *fp = fopen("input.txt", "r");

    // Variables to store grid dimensions
    int rows, cols;

    // Read rows and columns from file
    // Example:
    // 5 5
    fscanf(fp, "%d %d", &rows, &cols);

    // ------------------------------------------------------------
    // Dynamic 2D array allocation
    // ------------------------------------------------------------

    // grid stores current generation
    // next stores next generation

    int **grid = malloc(rows * sizeof(int *));
    int **next = malloc(rows * sizeof(int *));

    // Allocate memory for each row
    for (int i = 0; i < rows; i++)
    {
        // Allocate columns for current grid
        grid[i] = malloc(cols * sizeof(int));

        // Allocate columns for next generation grid
        next[i] = malloc(cols * sizeof(int));
    }

    // ------------------------------------------------------------
    // Read grid values from file
    // ------------------------------------------------------------
    for (int i = 0; i < rows; i++)
    {
        for (int j = 0; j < cols; j++)
        {
            fscanf(fp, "%d", &grid[i][j]);
        }
    }

    // Close file after reading
    fclose(fp);

    // ------------------------------------------------------------
    // Conway Simulation Loop
    // Runs for fixed number of generations
    // ------------------------------------------------------------
    for (int gen = 0; gen < GENERATIONS; gen++)
    {

        // --------------------------------------------------------
        // OpenMP Parallel Region
        //
        // parallel for:
        // Parallelizes loops
        //
        // collapse(2):
        // Combines ii and jj loops into one large loop
        //
        // schedule(static):
        // Divides iterations equally among threads
        // --------------------------------------------------------

#pragma omp parallel for collapse(2) schedule(static)

        // --------------------------------------------------------
        // TILE LOOP (ROW-WISE)
        //
        // ii represents tile starting row
        // Moves tile-by-tile instead of cell-by-cell
        // --------------------------------------------------------
        for (int ii = 0; ii < rows; ii += TILE)
        {

            // ----------------------------------------------------
            // TILE LOOP (COLUMN-WISE)
            //
            // jj represents tile starting column
            // ----------------------------------------------------
            for (int jj = 0; jj < cols; jj += TILE)
            {

                // ------------------------------------------------
                // Actual row loop inside tile
                // ------------------------------------------------
                for (int i = ii;
                     i < ii + TILE && i < rows;
                     i++)
                {

                    // --------------------------------------------
                    // Actual column loop inside tile
                    // --------------------------------------------
                    for (int j = jj;
                         j < jj + TILE && j < cols;
                         j++)
                    {

                        // ----------------------------------------
                        // Count alive neighbors around cell
                        // ----------------------------------------
                        int neighbors =
                            count_neighbors(
                                grid,
                                rows,
                                cols,
                                i,
                                j);

                        // ----------------------------------------
                        // Conway's Game of Life Rules
                        // ----------------------------------------

                        // If current cell is ALIVE
                        if (grid[i][j] == 1)
                        {

                            // Rule 1:
                            // Underpopulation (<2)
                            //
                            // Rule 2:
                            // Overpopulation (>3)
                            //
                            // In both cases cell dies

                            if (neighbors < 2 ||
                                neighbors > 3)
                                next[i][j] = 0;

                            // Otherwise survives
                            else
                                next[i][j] = 1;
                        }

                        // If current cell is DEAD
                        else
                        {

                            // Rule 3:
                            // Reproduction
                            //
                            // Dead cell with exactly
                            // 3 neighbors becomes alive

                            if (neighbors == 3)
                                next[i][j] = 1;

                            // Otherwise stays dead
                            else
                                next[i][j] = 0;
                        }
                    }
                }
            }
        }

        // --------------------------------------------------------
        // POINTER SWAP
        //
        // Instead of copying entire arrays,
        // simply swap pointers
        //
        // grid becomes next generation
        // next becomes old grid
        // --------------------------------------------------------

        int **temp = grid;
        grid = next;
        next = temp;
    }

    // ------------------------------------------------------------
    // Print final generation grid
    // ------------------------------------------------------------
    printf("Final Grid:\n");

    for (int i = 0; i < rows; i++)
    {
        for (int j = 0; j < cols; j++)
        {
            printf("%d ", grid[i][j]);
        }

        printf("\n");
    }

    // ------------------------------------------------------------
    // Free allocated memory
    // ------------------------------------------------------------

    for (int i = 0; i < rows; i++)
    {
        free(grid[i]);
        free(next[i]);
    }

    free(grid);
    free(next);

    return 0;
}

//wayoflifeMPI
#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>

#define GENERATIONS 10

// ------------------------------------------------------------
// Count neighbors (same logic as OpenMP version)
// grid is treated as 2D using grid[i][j]
// ------------------------------------------------------------
int count_neighbors(int **grid, int rows, int cols, int x, int y)
{
    int count = 0;

    for(int i = -1; i <= 1; i++)
    {
        for(int j = -1; j <= 1; j++)
        {
            if(i == 0 && j == 0)
                continue;

            int nx = x + i;
            int ny = y + j;

            if(nx >= 0 && nx < rows &&
               ny >= 0 && ny < cols)
            {
                count += grid[nx][ny];
            }
        }
    }

    return count;
}

// ------------------------------------------------------------
int main(int argc, char *argv[])
{
    MPI_Init(&argc, &argv);

    int rank, size;

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    int rows, cols;

    int **grid = NULL;   // full grid (only rank 0 uses initially)
    int **local;         // local chunk
    int **next;

    int *data = NULL;    // contiguous memory (IMPORTANT)

    // ------------------------------------------------------------
    // ROOT PROCESS: read file
    // ------------------------------------------------------------
    if(rank == 0)
    {
        FILE *fp = fopen("input.txt", "r");

        fscanf(fp, "%d %d", &rows, &cols);

        // contiguous allocation
        data = malloc(rows * cols * sizeof(int));

        // fill data
        for(int i = 0; i < rows; i++)
        {
            for(int j = 0; j < cols; j++)
            {
                fscanf(fp, "%d", &data[i * cols + j]);
            }
        }

        fclose(fp);
    }

    // ------------------------------------------------------------
    // Broadcast dimensions
    // ------------------------------------------------------------
    MPI_Bcast(&rows, 1, MPI_INT, 0, MPI_COMM_WORLD);
    MPI_Bcast(&cols, 1, MPI_INT, 0, MPI_COMM_WORLD);

    // ------------------------------------------------------------
    // Row decomposition (like tiling by rows)
    // ------------------------------------------------------------
    int local_rows = rows / size;

    // local contiguous block
    int *local_data = malloc(local_rows * cols * sizeof(int));
    int *next_data  = malloc(local_rows * cols * sizeof(int));

    // ------------------------------------------------------------
    // Scatter data
    // ------------------------------------------------------------
    MPI_Scatter(
        data,
        local_rows * cols,
        MPI_INT,
        local_data,
        local_rows * cols,
        MPI_INT,
        0,
        MPI_COMM_WORLD
    );

    // ------------------------------------------------------------
    // Convert 1D local_data into 2D pointer (for grid[i][j])
    // ------------------------------------------------------------
    local = malloc(local_rows * sizeof(int *));
    next  = malloc(local_rows * sizeof(int *));

    for(int i = 0; i < local_rows; i++)
    {
        local[i] = &local_data[i * cols];
        next[i]  = &next_data[i * cols];
    }

    // ------------------------------------------------------------
    // CONWAY SIMULATION
    // ------------------------------------------------------------
    for(int gen = 0; gen < GENERATIONS; gen++)
    {
        for(int i = 0; i < local_rows; i++)
        {
            for(int j = 0; j < cols; j++)
            {
                int neighbors = count_neighbors(local, local_rows, cols, i, j);

                if(local[i][j] == 1)
                {
                    if(neighbors < 2 || neighbors > 3)
                        next[i][j] = 0;
                    else
                        next[i][j] = 1;
                }
                else
                {
                    if(neighbors == 3)
                        next[i][j] = 1;
                    else
                        next[i][j] = 0;
                }
            }
        }

        // swap pointers
        int **temp = local;
        local = next;
        next = temp;
    }

    // ------------------------------------------------------------
    // Gather results back
    // ------------------------------------------------------------
    MPI_Gather(
        local_data,
        local_rows * cols,
        MPI_INT,
        data,
        local_rows * cols,
        MPI_INT,
        0,
        MPI_COMM_WORLD
    );

    // ------------------------------------------------------------
    // ROOT prints result
    // ------------------------------------------------------------
    if(rank == 0)
    {
        printf("Final Grid:\n");

        for(int i = 0; i < rows; i++)
        {
            for(int j = 0; j < cols; j++)
            {
                printf("%d ", data[i * cols + j]);
            }
            printf("\n");
        }
    }

    MPI_Finalize();
    return 0;
}

//ImageBlurMP
#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>

#define KERNEL 3   // 3x3 blur

// ----------------------------------------------------
// Compute average of neighbors (blur)
// ----------------------------------------------------
int get_blur(int *img, int rows, int cols, int x, int y)
{
    int sum = 0;
    int count = 0;

    for(int i = -1; i <= 1; i++)
    {
        for(int j = -1; j <= 1; j++)
        {
            int nx = x + i;
            int ny = y + j;

            if(nx >= 0 && nx < rows &&
               ny >= 0 && ny < cols)
            {
                sum += img[nx * cols + ny];
                count++;
            }
        }
    }

    return sum / count;
}

// ----------------------------------------------------
int main(int argc, char *argv[])
{
    MPI_Init(&argc, &argv);

    int rank, size;
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    int rows, cols;

    int *img = NULL;

    // ---------------- ROOT READ FILE ----------------
    if(rank == 0)
    {
        FILE *fp = fopen("image.txt", "r");

        fscanf(fp, "%d %d", &rows, &cols);

        img = malloc(rows * cols * sizeof(int));

        for(int i = 0; i < rows; i++)
        {
            for(int j = 0; j < cols; j++)
            {
                fscanf(fp, "%d", &img[i * cols + j]);
            }
        }

        fclose(fp);
    }

    // broadcast size
    MPI_Bcast(&rows, 1, MPI_INT, 0, MPI_COMM_WORLD);
    MPI_Bcast(&cols, 1, MPI_INT, 0, MPI_COMM_WORLD);

    int local_rows = rows / size;

    int *local = malloc(local_rows * cols * sizeof(int));
    int *result = malloc(local_rows * cols * sizeof(int));

    // distribute image rows
    MPI_Scatter(img,
                local_rows * cols,
                MPI_INT,
                local,
                local_rows * cols,
                MPI_INT,
                0,
                MPI_COMM_WORLD);

    // ---------------- PROCESSING ----------------
    for(int i = 0; i < local_rows; i++)
    {
        for(int j = 0; j < cols; j++)
        {
            result[i * cols + j] =
                get_blur(local, local_rows, cols, i, j);
        }
    }

    // gather results
    MPI_Gather(result,
               local_rows * cols,
               MPI_INT,
               img,
               local_rows * cols,
               MPI_INT,
               0,
               MPI_COMM_WORLD);

    // print result
    if(rank == 0)
    {
        printf("Blurred Image:\n");

        for(int i = 0; i < rows; i++)
        {
            for(int j = 0; j < cols; j++)
            {
                printf("%d ", img[i * cols + j]);
            }
            printf("\n");
        }
    }

    MPI_Finalize();
    return 0;
}

//ImageBlurOpenMP
#include <stdio.h>
#include <stdlib.h>
#include <omp.h>

#define TILE 32

// ------------------------------------------------------------
// Blur function
// Computes average of neighboring pixels
// ------------------------------------------------------------
int blur_pixel(int **img,
               int rows,
               int cols,
               int x,
               int y)
{
    int sum = 0;
    int count = 0;

    // Traverse neighboring cells
    for(int i=-1; i<=1; i++)
    {
        for(int j=-1; j<=1; j++)
        {
            int nx = x + i;
            int ny = y + j;

            // Boundary check
            if(nx>=0 && nx<rows &&
               ny>=0 && ny<cols)
            {
                sum += img[nx][ny];
                count++;
            }
        }
    }

    // Return average
    return sum / count;
}

int main()
{
    FILE *fp = fopen("image.txt","r");

    int rows, cols;

    fscanf(fp,"%d %d",&rows,&cols);

    // --------------------------------------------------------
    // Dynamic 2D allocation
    // --------------------------------------------------------
    int **img = malloc(rows*sizeof(int *));
    int **result = malloc(rows*sizeof(int *));

    for(int i=0; i<rows; i++)
    {
        img[i] = malloc(cols*sizeof(int));
        result[i] = malloc(cols*sizeof(int));
    }

    // Read image
    for(int i=0; i<rows; i++)
    {
        for(int j=0; j<cols; j++)
        {
            fscanf(fp,"%d",&img[i][j]);
        }
    }

    fclose(fp);

    // --------------------------------------------------------
    // Parallel tiled blur
    // --------------------------------------------------------
    #pragma omp parallel for collapse(2) schedule(static)

    for(int ii=0; ii<rows; ii+=TILE)
    {
        for(int jj=0; jj<cols; jj+=TILE)
        {
            for(int i=ii;
                i<ii+TILE && i<rows;
                i++)
            {
                for(int j=jj;
                    j<jj+TILE && j<cols;
                    j++)
                {
                    result[i][j] =
                        blur_pixel(
                            img,
                            rows,
                            cols,
                            i,
                            j
                        );
                }
            }
        }
    }

    // Print blurred image
    printf("Blurred Image:\n");

    for(int i=0; i<rows; i++)
    {
        for(int j=0; j<cols; j++)
        {
            printf("%d ", result[i][j]);
        }

        printf("\n");
    }

    return 0;
}

//OPENMP MATRIX MULTIPLICATION
#include <stdio.h>
#include <stdlib.h>
#include <omp.h>

#define TILE 32

int main()
{
    FILE *fp = fopen("matrix.txt","r");

    int rowsA, colsA, colsB;

    // Read dimensions
    fscanf(fp,"%d %d %d",
           &rowsA,
           &colsA,
           &colsB);

    // --------------------------------------------------------
    // Dynamic allocation
    // --------------------------------------------------------
    int **A = malloc(rowsA*sizeof(int *));
    int **B = malloc(colsA*sizeof(int *));
    int **C = malloc(rowsA*sizeof(int *));

    for(int i=0; i<rowsA; i++)
    {
        A[i] = malloc(colsA*sizeof(int));
        C[i] = malloc(colsB*sizeof(int));
    }

    for(int i=0; i<colsA; i++)
    {
        B[i] = malloc(colsB*sizeof(int));
    }

    // --------------------------------------------------------
    // Read Matrix A
    // --------------------------------------------------------
    for(int i=0; i<rowsA; i++)
    {
        for(int j=0; j<colsA; j++)
        {
            fscanf(fp,"%d",&A[i][j]);
        }
    }

    // --------------------------------------------------------
    // Read Matrix B
    // --------------------------------------------------------
    for(int i=0; i<colsA; i++)
    {
        for(int j=0; j<colsB; j++)
        {
            fscanf(fp,"%d",&B[i][j]);
        }
    }

    fclose(fp);

    // --------------------------------------------------------
    // Initialize result matrix
    // --------------------------------------------------------
    for(int i=0; i<rowsA; i++)
    {
        for(int j=0; j<colsB; j++)
        {
            C[i][j] = 0;
        }
    }

    // --------------------------------------------------------
    // Parallel tiled matrix multiplication
    // --------------------------------------------------------
    #pragma omp parallel for collapse(2) schedule(static)

    for(int ii=0; ii<rowsA; ii+=TILE)
    {
        for(int jj=0; jj<colsB; jj+=TILE)
        {

            for(int kk=0; kk<colsA; kk+=TILE)
            {

                for(int i=ii;
                    i<ii+TILE && i<rowsA;
                    i++)
                {

                    for(int j=jj;
                        j<jj+TILE && j<colsB;
                        j++)
                    {

                        for(int k=kk;
                            k<kk+TILE && k<colsA;
                            k++)
                        {

                            C[i][j] +=
                                A[i][k] * B[k][j];
                        }
                    }
                }
            }
        }
    }

    // --------------------------------------------------------
    // Print result matrix
    // --------------------------------------------------------
    printf("Result Matrix:\n");

    for(int i=0; i<rowsA; i++)
    {
        for(int j=0; j<colsB; j++)
        {
            printf("%d ", C[i][j]);
        }

        printf("\n");
    }

    return 0;
}

//MPI MATRIX MULTIPLICATION
#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>

// ----------------------------------------------------
int main(int argc, char *argv[])
{
    MPI_Init(&argc, &argv);

    int rank, size;

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    int A_rows, A_cols, B_cols;

    int *A = NULL;
    int *B = NULL;
    int *C = NULL;

    // ---------------- ROOT READ FILE ----------------
    if(rank == 0)
    {
        FILE *fp = fopen("matrix.txt", "r");

        fscanf(fp, "%d %d %d", &A_rows, &A_cols, &B_cols);

        A = malloc(A_rows * A_cols * sizeof(int));
        B = malloc(A_cols * B_cols * sizeof(int));
        C = malloc(A_rows * B_cols * sizeof(int));

        for(int i = 0; i < A_rows; i++)
        {
            for(int j = 0; j < A_cols; j++)
            {
                fscanf(fp, "%d", &A[i * A_cols + j]);
            }
        }

        for(int i = 0; i < A_cols; i++)
        {
            for(int j = 0; j < B_cols; j++)
            {
                fscanf(fp, "%d", &B[i * B_cols + j]);
            }
        }

        fclose(fp);
    }

    // broadcast sizes
    MPI_Bcast(&A_rows, 1, MPI_INT, 0, MPI_COMM_WORLD);
    MPI_Bcast(&A_cols, 1, MPI_INT, 0, MPI_COMM_WORLD);
    MPI_Bcast(&B_cols, 1, MPI_INT, 0, MPI_COMM_WORLD);

    int local_rows = A_rows / size;

    int *local_A = malloc(local_rows * A_cols * sizeof(int));
    int *local_C = malloc(local_rows * B_cols * sizeof(int));

    // send matrix B to all processes
    B = malloc(A_cols * B_cols * sizeof(int));
    MPI_Bcast(B, A_cols * B_cols, MPI_INT, 0, MPI_COMM_WORLD);

    // scatter A
    MPI_Scatter(A,
                local_rows * A_cols,
                MPI_INT,
                local_A,
                local_rows * A_cols,
                MPI_INT,
                0,
                MPI_COMM_WORLD);

    // ---------------- MATRIX MULTIPLICATION ----------------
    for(int i = 0; i < local_rows; i++)
    {
        for(int j = 0; j < B_cols; j++)
        {
            local_C[i * B_cols + j] = 0;

            for(int k = 0; k < A_cols; k++)
            {
                local_C[i * B_cols + j] +=
                    local_A[i * A_cols + k] * B[k * B_cols + j];
            }
        }
    }

    // gather result
    MPI_Gather(local_C,
               local_rows * B_cols,
               MPI_INT,
               C,
               local_rows * B_cols,
               MPI_INT,
               0,
               MPI_COMM_WORLD);

    // print result
    if(rank == 0)
    {
        printf("Result Matrix:\n");

        for(int i = 0; i < A_rows; i++)
        {
            for(int j = 0; j < B_cols; j++)
            {
                printf("%d ", C[i * B_cols + j]);
            }
            printf("\n");
        }
    }

    MPI_Finalize();
    return 0;
}
